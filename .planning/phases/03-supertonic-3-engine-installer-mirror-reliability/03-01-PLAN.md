class SubprocessBackend(TTSBackend):
    """Base class. Manages: venv selection, Popen lifecycle (long-lived sidecar),
    length-prefixed JSON-over-stdio IPC, atexit SIGTERM, HF_HOME/HF_TOKEN/HF_ENDPOINT
    /HF_HUB_CACHE inheritance via env=os.environ.copy() at Popen time.

    Subclasses set:
        - id: str
        - display_name: str
        - sidecar_module: str       # e.g. "engines.supertonic3.sidecar"
        - venv_path: Path | None    # optional dedicated venv; None = main venv
        - extra_env: dict[str, str] # merged on top of os.environ.copy()

    Subclasses call:
        - self._sub.invoke({"op": "synthesize", ...}) -> dict  (sync IPC)
        - self._sub.start()  (idempotent; lazy on first generate())
        - self._sub.health_check() -> bool
        - self._sub.shutdown()  (atexit-registered automatically)
    """
